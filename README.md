<!--eslint-skip-->

```jsonc
// .vscode/setting.json
{
  "window.autoDetectColorScheme": true,
  "workbench.preferredLightColorTheme": "oQ Light",
  "workbench.preferredDarkColorTheme": "oQ Dark",
}
```
